models = [lr, knn, svm, nb, dt, rf]
scores =[]

for model in models:
    model.fit(X_train,y_train)
    y_pred =model.predict(X_test)
    scores.append(accuracy_score(y_test ,y_pred))
    print("Accuracy of "+ type(model).__name__ + "is" ,accuracy_score(y_test,y_pred))
